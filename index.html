// ========================================== // FUNCIÓN LECTORA (Devuelve Catálogo y Consecutivo) // ========================================== function doGet(e) { var doc = SpreadsheetApp.getActiveSpreadsheet(); // 1. Obtener consecutivo var sheetVentas = doc.getSheetByName("Ventas"); var nextNumber = 1; if (sheetVentas && sheetVentas.getLastRow() > 1) { var values = sheetVentas.getRange(2, 2, sheetVentas.getLastRow() - 1, 1).getValues(); for (var i = 0; i < values.length; i++) { var profStr = String(values[i][0]); var match = profStr.match(/(\d+)$/); if (match) { var num = parseInt(match[1], 10); if (num >= nextNumber) { nextNumber = num + 1; } } } } // 2. Obtener Catálogo completo var sheetProd = doc.getSheetByName("Productos"); var products = []; if (sheetProd && sheetProd.getLastRow() > 1) { // Lee desde la fila 2, columnas A, B y C var prodValues = sheetProd.getRange(2, 1, sheetProd.getLastRow() - 1, 3).getValues(); for (var j = 0; j < prodValues.length; j++) { var desc = String(prodValues[j][0] || "").trim(); if (desc !== "") { products.push({ desc: desc, cost: parseFloat(prodValues[j][1]) || 0, price: parseFloat(prodValues[j][2]) || 0 }); } } } return ContentService.createTextOutput(JSON.stringify({ next: nextNumber, products: products })).setMimeType(ContentService.MimeType.JSON); } // ========================================== // FUNCIÓN ESCRITORA (Guarda arriba y limpia formato) // ========================================== function doPost(e) { var doc = SpreadsheetApp.getActiveSpreadsheet(); var rawData = JSON.parse(e.postData.contents); if (rawData.type === "producto") { var sheetProd = doc.getSheetByName("Productos"); if (!sheetProd) { sheetProd = doc.insertSheet("Productos"); sheetProd.appendRow(["Descripción", "Costo Unitario (S/)", "Precio Venta (S/)"]); sheetProd.getRange(1, 1, 1, 3).setFontWeight("bold").setBackground("#fff2cc"); } sheetProd.insertRowAfter(1); var newRow = sheetProd.getRange(2, 1, 1, 3); newRow.setValues([[rawData.data.Descripcion, rawData.data.Costo, rawData.data.Precio]]); newRow.clearFormat(); return ContentService.createTextOutput("Producto agregado"); } var data = rawData; if(!data || data.length === 0) return ContentService.createTextOutput("Sin datos"); var sheetVentas = doc.getSheetByName("Ventas"); if (!sheetVentas) { sheetVentas = doc.getSheets()[0]; if (sheetVentas.getName() !== "Ventas") { sheetVentas.setName("Ventas"); } } var headers = ["Fecha", "Numero_Proforma", "Cliente", "Estado", "Modo_Vista", "Desc_General_Combo", "Item_Cant", "Item_Descripcion", "Item_Costo_U", "Item_Precio_U", "Item_Importe_Total", "Item_Ganancia", "Proforma_Inversion_Total", "Proforma_Venta_Total", "Proforma_Ganancia_Neta"]; if (sheetVentas.getLastRow() === 0) { sheetVentas.appendRow(headers); sheetVentas.getRange(1, 1, 1, headers.length).setFontWeight("bold").setBackground("#cfe2f3"); } var newValues = []; for (var i = 0; i < data.length; i++) { var row = []; for (var j = 0; j < headers.length; j++) { row.push(data[i][headers[j]]); } newValues.push(row); } sheetVentas.insertRowsAfter(1, newValues.length); var newRange = sheetVentas.getRange(2, 1, newValues.length, headers.length); newRange.setValues(newValues); newRange.clearFormat(); return ContentService.createTextOutput("Proforma agregada"); }
